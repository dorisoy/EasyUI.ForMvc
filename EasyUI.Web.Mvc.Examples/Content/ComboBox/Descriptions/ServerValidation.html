This example shows how EasyUI ComboBox for ASP.NET MVC is being validated through the built-in server-side validation capabilities of ASP.NET MVC. 


<%= Html.ValidationSummary() %>
<% using (Html.BeginForm("Action", "Controller")) { %>
    <%= Html.EasyUI()
            .ComboBox()
            .Name("Products")
    %>
    <%= Html.ValidationMessage("Products", "*") %>
    <input type="submit" value="Post" />
<% } %>
